          var viewModel = {};
          //Initializing Model

          viewModel.Min = ko.observable().extend({ min: { params: 20, message: "Age Must be greater than 20" }, required: { params: true, message: "Please enter value" } });
          // Adding Min property in the ViewModel and Validate number must be greater than 20.

          viewModel.rangelength = ko.observable().extend({ min: { params: 5, message: "value should be greater than 5" }, max: { params: 50, message: "value must be less than 50." } });
          // Adding rangelength property in the ViewModel and validate its range 5 to 50 .

          viewModel.date = ko.observable().extend({ date: { params: true, message: "Please enter a valid date" }, required: { params: true, message: "Please enter date" } });
          // Adding date property in the ViewModel and applying date format validation.

          viewModel.number = ko.observable().extend({ number: { params: true, message: "only numbers are allowed" }, required: { params: true, message: "Please enter number" } });
          // Adding Number property in the ViewModel and apply Number Validation.

          viewModel.Digits = ko.observable().extend({ minLength: { digit: true, message: "only digits are allowed" }, required: { params: true, message: "Please enter digit" } });
          // Adding Digits property in the ViewModel and Make it observalble which is responsible for dual binding.

          viewModel.Gender = ko.observable().extend({ required: { params: true, message: "Gender required" } });
          // Adding Gender property in the ViewModel and Make it required field.

          viewModel.Phone = ko.observable().extend({ phoneUS: { params: true, message: "please enter phone in US format" } });
          // Adding Phone property in the ViewModel and Apply US format validation.

          viewModel.Approve = ko.observableArray(["Cricket"]).extend({
           required: { params: true, message: "please select at least one" }
           });
          // Adding Approve property in the ViewModel which is an Array by which User can select more than one hobbies through te checkbox selection.

          viewModel.Address = ko.observable().extend({ required: { params: true, message: "Where is your residence" } });
          // Adding Address property in the ViewModel and Make it observalble which is responsible for dual binding.

          ko.validation.group(viewModel, { deep: true });
          // Groupng all the property and pass the property true which means to validate the model.

          ko.applyBindings(viewModel);
          // Applying the Binding on the Model
