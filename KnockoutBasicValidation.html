          var viewModel = {};
          //Initializing Model

          viewModel.Name = ko.observable().extend({ required: { params: true, message: "Please enter User Name" }, minLength: { params: 6, message: "ohhhh!!!!! come on ,Please enter altleast 6 characters" } });
          // Adding Name propery in the ViewModel and Make it observalble which is responsible for dual binding.

          viewModel.Email = ko.observable().extend({ email: { params: true, message: "Please enter valid Email" }, required: { params: true, message: "email required" } });
          // Adding Email propery in the ViewModel and forcely validate to email type to user.

          viewModel.Password = ko.observable().extend({ pattern: { params: '^([a-zA-Z0-9 _-]+)$', message: "Only Number are required." }, required: { params: true, message: "password required" } });
         // Adding Password propery in the ViewModel and passing the regex(AlphaNumeric) , when user's input match the pattern then this user will be validate.

          viewModel.Degree = ko.observable('').extend({ required: { params: true, message: "Degree required" } });
         // Adding Degree propery in the ViewModel and Make it observalble which is responsible for dual binding.

         viewModel.Description = ko.observable().extend({ minLength: { params: 20, message: "ohhhhh!!!!! come on you have to write at least 20 words" }, maxLength: { params: 50, message: "I think 50 words are sufficient." } });
         // Adding Description propery in the ViewModel and describe the minimum maximum length of the text.

          ko.validation.group(viewModel, { deep: true });
          // Groupng all the property and pass the property true which means to validate the model.

          ko.applyBindings(viewModel);
          // Applying the Binding on the Model
